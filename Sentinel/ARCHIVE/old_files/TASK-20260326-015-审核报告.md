```python
def _check_eastmoney_api(self) -> bool:
    """检查东方财富API"""
    try:
        import requests
        from requests.adapters import HTTPAdapter
        from urllib3.util.retry import Retry

        # 配置重试策略
        retry_strategy = Retry(
            total=3,
            backoff_factor=1,
            status_forcelist=[429, 500, 502, 503, 504]

        adapter = HTTPAdapter(max_retries=retry_strategy)
        session = requests.Session()

            "https://82.push2.eastmoney.com/api/qt/clist/get",
            params={
                'pn': 1,
                'pz': 10,
```

```python
def _check_data_freshness(self) -> bool:
    """检查数据新鲜度"""
    logger.info("\n【2】检查数据新鲜度...")

    all_ok = True

    # 定义不同步骤的数据新鲜度阈值（小时）
    freshness_thresholds = {
        'Step1 数据层': 24,
        'Step2 因子层': 24,
        'Step3 策略层': 24,
        'Step4 组合优化': 24,
        'Step5 风控层': 24,
        'Step6 交易层': 24,
    }

    for step in self.manager.STEP_ORDER:
        record = self.manager.get_step_record(step)
        if record and record.completed_at:
            freshness_hours = record.data_freshness_hours
            logger.info(f"  {step}: {freshness_hours:.1f} 小时前")

            # 检查是否过期
            threshold = freshness_thresholds.get(step, 24)
            if freshness_hours > threshold:
                logger.warning(f"  ⚠️ {step} 数据过期（超过{threshold}小时）")
                self.issues.append({
                    'type': 'data_freshness',
                    'step': step,
                    'severity': 'high',
                    'description': f'{step} 数据过期（{freshness_hours:.1f}小时）',
                    'impact': '可能导致推送信息不准确'
                })
                all_ok = False

    return all_ok
```
